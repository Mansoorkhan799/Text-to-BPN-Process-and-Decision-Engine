\section{Python Code Listing}
\begin{lstlisting}[language=Python, caption=Simple Python Example]
def factorial(n):
    """Compute factorial using recursion."""
    if n == 0:
        return 1
    else:
        return n * factorial(n - 1)

print(factorial(5))
\end{lstlisting}
